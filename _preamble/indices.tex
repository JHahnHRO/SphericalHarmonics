%%% Index% for multiple indices\usepackage{imakeidx}\indexsetup{	level=\chapter*,                % Headings of indices will be displayed as \chapter* would	toclevel=chapter,               % Indices will appear in the table of contents like a chapter	headers={\indexname}{\indexname}% Header will contain the name	}% creates a new index\makeindex[	% internal name of the index	name=terms,	% Displayed name	title=Stichwortverzeichnis,	% Will be listed in the table of contents	intoc,	% will be displayed in two columns	columns=2,	% will be generated automatically with makeindex and the given program options inlcuding the	% *.ist file (index style) that is to be used.	program=makeindex,options=-s terms.ist	]
